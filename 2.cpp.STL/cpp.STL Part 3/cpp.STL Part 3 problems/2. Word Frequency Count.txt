//2.Word Frequency Count 

//input 
Paragraph: "the quick brown fox jumps over the lazy dog the quick fox" 

//output 
the - 3
quick - 2
brown - 1
fox - 2
jumps - 1
over - 1
lazy - 1
dog - 1 

//solution 

#include<bits/stdc++.h>
using namespace std;

class solution{
    public:
    void countWords(map<string, int>& wordCount, const string& paragraph) {
        stringstream ss(paragraph); 
        string word; 
        while(ss>>word){
            wordCount[word]++;
        }
        
    }
    
    void displayWordCount(const map<string, int>& wordCount) {
        for(const auto& pair : wordCount){
            cout <<pair.first <<"-"<<pair.second<< endl;
        }
    }
};
