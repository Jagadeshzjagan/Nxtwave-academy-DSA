//vector 
  Vectors are like dynamic arrays, automatically resizing as elements are added or
  removed with contiguous storage for easy access

  To use the vector library in C++, include the <vector> header and can also be used by adding <bits/stdc++.h>

//example 1 
#include <iostream>
#include <vector>
// or use #include <bits/stdc++.h>
using namespace std;


int main() {
    vector<int> v;
    v.push_back(10); 
    v.emplace_back(20);
    cout << v[0] << " " << v[1] << endl; // Output - 10 20
    return 0;
} 

//output 
10 20 

//example 2 
#include <iostream>
#include <vector>
// or use #include <bits/stdc++.h>
using namespace std;

int main() {
    // 1. Simple Vector Declaration
    vector<int> v;
    // 2. Vector Initialization with Size and Default Value
    vector<int> v1(5, 20);
    // 3. Vector Initialization with Size Only
    vector<int> v2(5);
    // 4. Copying One Vector to Another
    vector<int> v3(v1);
    return 0;
}
