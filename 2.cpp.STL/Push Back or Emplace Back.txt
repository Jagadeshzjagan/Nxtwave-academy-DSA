//Push Back or Emplace Back

//example 1 
#include <iostream>
#include <vector> // or use #include <bits/stdc++.h>
using namespace std;

int main() {
    vector<int> v;
    v.push_back(10); 
    v.emplace_back(20);
    cout << v[0] << " " << v[1] << endl; // Output - 10 20
    return 0;
} 

//vector<int> v Declares a dynamic integer vector named 'v'

//v.push_back(10) Adds 10 to the end of 'v' using the 'push_back' method

//v.emplace_back(20) Adds 20 to 'v' using 'emplace_back', constructing it in place

//cout << v[0] << " " << v[1] << endl Prints the values in 'v' (10 and 20)
  with a space between them, followed by a newline character
