//7.Set
Set is a container that stores unique elements in a sorted order. 
It doesn't allow duplicate values.

Properties
Sorted :  Yes, the elements are stored in a sorted order.

Unique :  Yes, it allows only unique elements.

//example 
basic operation 

#include <iostream>
#include <set>
using namespace std;

int main() {
    // Create a set and insert elements
    set<int> s;
    
    s.insert(1); // {1}
    s.insert(3); // {1, 3}
    s.insert(5); // {1, 3, 5}
    s.insert(7); // {1, 3, 5, 7}
    s.insert(3); // {1, 3, 5, 7} --> 3 doesn't get inserted
    
    // Erase elements from the set
    s.erase(3); // Erase element '3'
    
    auto it7 = s.find(7); // Find iterator for element '7'
    s.erase(it7); // Erase element '7' using iterator

    s = {2, 4, 6, 8, 10};

    int count8 = s.count(8);
    int count5 = s.count(5);
    cout << "Count of 8 in the set: " << count8 << endl; // 1
    cout << "Count of 5 in the set: " << count5 << endl; // 0
    return 0;
}


//Explanation :

//set<int> s :  Declare a set for storing unique integers.

//s.insert() :  Insert elements into the set.

//auto it5 = s.find(5) and auto it9 = s.find(9) :  Find elements in the set using find and assign iterators to them.

//s.erase(3) and s.erase(it7) :  Erase elements from the set by value or using iterators.

//s.count(8) and s.count(5) :  Count the occurrences of specific values in the set using count
