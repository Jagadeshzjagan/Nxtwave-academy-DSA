#include <bits/stdc++.h>
using namespace std;

class solution{
    public:
    void operate( vector<pair<int, string>> &pairs) {
        int sum= 0; 
        string concat=""; 
        for(auto p : pairs){
            sum+=p.first; 
            concat+=p.second;
        }
        cout<<sum<<endl;
        cout<<concat<<endl;
        cout<<concat.length()<<endl;
    }
};
