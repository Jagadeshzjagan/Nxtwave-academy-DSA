//solution 
class solution {
public:
    void printAscLetterPyramid(int n) {
        for (int i=1; i<=n; i++){
            for (int j=i-1; j>=0; j--){
                cout<<char('A'+ n-j-1)<<" "; 
            }cout<<endl;
        }   
    }
};
