class solution {
public:
    void printTrianglePattern(int n) {
        for (int i=1; i<n+1; i++){
            for (int j=0; j<i; j++){cout<<"* ";
            }cout<<endl;
        }
        for(int i=1; i<n; i++){
            for (int j=0+i; j<n; j++){ cout<<"* ";
            }cout<<endl;
        }
    }
};
