//solution  
class solution {
public:
    void printAscLetterTriangle(int n) {
        for (int i=0; i<n; i++){
            for (int j=0; j<=i; j++){
                cout<<char('A'+i)<<" "; 
            }cout<<endl;
        }
    }
};
